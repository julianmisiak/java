CREATE TABLE USUARIO
(
OID_USUARIO NUMBER(10),
OID_ROL NUMBER(10),
USUARIO VARCHAR2(25) NOT NULL,
PASS VARCHAR(25) NOT NULL, 
DNI NUMBER NOT NULL,
NOMBRE VARCHAR2(25) NOT NULL,
APELLIDO VARCHAR(25) NOT NULL,
ESTADO NUMBER(1) NOT NULL
);

ALTER TABLE USUARIO
ADD(
CONSTRAINT OID_PK_USER PRIMARY KEY (OID_USUARIO),
CONSTRAINT OID_FK FOREIGN KEY (OID_ROL) REFERENCES ROL_TBL(OID_ROL)
);

CREATE TABLE ROL_TBL
(
OID_ROL NUMBER(10),
TIPO NUMBER NOT NULL,
DESCRIPCION VARCHAR2(25) NOT NULL
);

ALTER TABLE ROL_TBL
ADD(
CONSTRAINT OID_PK PRIMARY KEY (OID_ROL)
);
